Ejecuta una sentencia select para todos los atributos necesarios para tener en detalle de un Producto. Recibe el ID del Producto que estamos buscando.

LEFT JOIN EMPSERVPROD AS ESP ON P.TPRODID = ESP.TPRODID4 LEFT JOIN EMPSERV AS ES ON ES.TESID = ESP.TESID1 LEFT JOIN SERVICIOS AS SERV ON SERV.TSERVID = ES.TSERVID1 LEFT JOIN EMPRESAS AS EMP ON EMP.TEMPID = ES.TEMP_ID1
